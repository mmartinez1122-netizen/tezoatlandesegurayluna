document.addEventListener('DOMContentLoaded', () => {
    const timelineItems = document.querySelectorAll('.timeline-item');

    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                entry.target.classList.add('visible');
                // Opcional: Desobservar una vez que se activa la animación
                // observer.unobserve(entry.target); 
            }
        });
    }, {
        threshold: 0.2 // Activar cuando el 20% del elemento sea visible
    });

    timelineItems.forEach(item => {
        item.classList.add('fade-in'); // Añade clase inicial para animación
        observer.observe(item);
    });

    console.log('history.js cargado. Animaciones de línea de tiempo listas.');
});
document.addEventListener('DOMContentLoaded', () => {
    const detailsElements = document.querySelectorAll('details');

    detailsElements.forEach(details => {
        details.addEventListener('toggle', (event) => {
            const summary = details.querySelector('summary');
            if (details.open) {
                summary.textContent = 'Ocultar información'; // Cambia el texto al abrir
            } else {
                summary.textContent = 'Ver información'; // Restaura el texto al cerrar
            }
        });
        
        // Asegura que el texto inicial del summary sea el correcto
        const summary = details.querySelector('summary');
        if (details.open) {
            summary.textContent = 'Ocultar información';
        } else {
            summary.textContent = 'Ver información';
        }
    });

    console.log('culture.js cargado. Funcionalidad de detalles/summary activa.');
});
